199int DrawLine(int x1, int x2, int y1, int y2, int color)
200{
201 if(x1 > x2) // will be optimised out
202 {
203 int xHolder = x1;
204 x1 = x2;
205 x2 = xHolder;
206 }
207 if(y1 > y2) // will be optimised out
208 {
209 int yHolder = y1;
210 y1 = y2;
211 y2 = yHolder;
212 }
213 int deltaX = x2 - x1;
214 int deltaY = y2 - y1;
215 int errorXY = 0;
216 if(deltaX == 0)
217 {
218 errorXY = INT32_MIN;
219 }
220 else if(deltaY == 0)
221 {
222 errorXY = INT32_MAX;
223 }
224 else
225 {
226 errorXY = (deltaX<<SHIFTING_MULTIPLY_FLOATLESS_CALC) / deltaY;
227 }
228 int workingError = errorXY;
229 while((x1 != x2) || (y1 != y2))
230 {
231 VGA_SetPixel(x1, y1, color);
232 if(workingError == INT32_MAX)
233 {
234 x1++;
235 continue;
236 }
237 else if(workingError == INT32_MIN)
238 {
239 y1++;
240 continue;
241 }
242 if(workingError == (1<<SHIFTING_MULTIPLY_FLOATLESS_CALC))
243 {
244 x1++;
245 y1++;
246 workingError -= (1<<SHIFTING_MULTIPLY_FLOATLESS_CALC);
247 }
248 else if(workingError > (1<<SHIFTING_MULTIPLY_FLOATLESS_CALC))
249 {
250 x1++;
251 workingError -= (1<<SHIFTING_MULTIPLY_FLOATLESS_CALC);
252 continue;
253 }
254 else
255 {
256 y1++;
257 }
258 workingError += errorXY;
259 }
260 return 0;
261}
