41void VGA_Init(void)
42{
43 vgaData_s.lineCounter = 0;
44 vgaData_s.startAddressDMA = 0;
45 vgaData_s.dmaConfigRegister = 0;
46
47 GPIOB->BSRR = VGA_VSYNC_Pin;
48
49 // TIM2
50 HAL_TIM_Base_Start(&htim2);
51 HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_4);
52 HAL_TIM_PWM_Start_IT(&htim2, TIM_CHANNEL_3);
53 // TIM1
54 __HAL_TIM_ENABLE_DMA(&htim1, TIM_DMA_UPDATE);
55 __HAL_TIM_ENABLE(&htim1);
56 HAL_DMA_Start_IT(&hdma_tim1_up, (uint32_t)&vgaData_s.videoRAM[0], VGA_GPIOE_ODR_ADDRESS, VGA_DISPLAY_X + 1);
57
58 HAL_DMA_Init(&hdma_tim1_up);
59 __HAL_DMA_ENABLE_IT(&hdma_tim1_up, DMA_IT_TC);
60 //-----------------------
61 // Register swap and safe
62 //-----------------------
63 // content of CR-Register read and save
64 vgaData_s.dmaConfigRegister = DMA2_Stream5->CR;
65 LOGI("VGA initialised");
66}

89__inline void VGA_InterruptHsync(void)
90{
91 vgaData_s.lineCounter++;
92 if(vgaData_s.lineCounter >= VGA_VSYNC_PERIODE)
93 {
94 // Address pointer first dot
95 vgaData_s.lineCounter = 0;
96 vgaData_s.startAddressDMA = (uint32_t)(&vgaData_s.videoRAM[0]);
97 }
98
99 // HSync-Pixel
100 GPIOB->BSRR = (vgaData_s.lineCounter < VGA_VSYNC_IMP) ? VGA_VSYNC_Pin<<16u : VGA_VSYNC_Pin;
101
102 // Test for DMA start
103 if((vgaData_s.lineCounter >= VGA_VSYNC_BILD_START) && (vgaData_s.lineCounter <= VGA_VSYNC_BILD_STOP))
104 {
105 // Restart DMA transfer
106 DMA2_Stream5->CR = vgaData_s.dmaConfigRegister;
107 DMA2_Stream5->M0AR = vgaData_s.startAddressDMA;
108 TIM1->CR1 |= TIM_CR1_CEN;
109 __HAL_DMA_ENABLE(&hdma_tim1_up);
110 // Test Adrespointer for high
111 if(vgaData_s.lineCounter&0x01)
112 vgaData_s.startAddressDMA += (VGA_DISPLAY_X + 1); // inc after Hsync
113 }
114}
